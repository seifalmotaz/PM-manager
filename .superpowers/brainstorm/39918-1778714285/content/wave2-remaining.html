Wave 2: Notifications + Employee Pages + Forecasting

notifications Table

id uuid PK
userId uuid FK → users // who receives this
type text // 'assigned' | 'deadline_soon' | 'sprint_started' | 'sprint_ended' | 'mentioned'
title text NOT NULL // "Assigned to 'Fix login bug'"
body text
entityType text // task | sprint | project
entityId uuid // click → opens entity
isRead boolean DEFAULT false
createdAt timestamp
Index: (userId, isRead, createdAt) for unread-first queries
